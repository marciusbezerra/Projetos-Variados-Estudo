FUNCTION JANELA( PJAN1, PJAN2, PJAN3, PJAN4, PJAN5 )
IF PCOUNT() != 5
   PJAN5 := ""
ENDIF
SOMBRA( PJAN1, PJAN2, PJAN3, PJAN4 )
SETCOLOR(  "07/01" )
@ PJAN1, PJAN2 CLEAR TO PJAN3, PJAN4
SETCOLOR( "14/01" )
@ PJAN1, PJAN2 to PJAN3, PJAN4 DOUBLE
SETCOLOR( "01/27" )
@ PJAN1, PJAN2+1 SAY space(PJAN4-PJAN2-1)
IF LEN( TRIM( PJAN5 ) ) > 0
   @ PJAN1, PJAN2 + ( ( ( PJAN4 + 1 - PJAN2 ) - LEN( PJAN5 ) ) / 2 ) SAY PJAN5
ENDIF
